\section*{}
\noindent Consider the matrix
\[
A = \begin{bmatrix}
3 & 0 & 0 \\
0 & 1 & 1 \\
0 & 1 & 1
\end{bmatrix}.
\]
\subsection*{(a) Characteristic Polynomial}
Determine the characteristic polynomial $\det(A - \lambda I_3)$ of $A$.

\begin{equation*}
    A-\lambda I_3=
    \begin{bmatrix}
3 & 0 & 0 \\
0 & 1 & 1 \\
0 & 1 & 1
\end{bmatrix}-
    \begin{bmatrix}
        \lambda  & 0 & 0 \\
        0 & \lambda & 0 \\
        0 & 0 & \lambda
    \end{bmatrix} =
  \begin{bmatrix}
        3 - \lambda & 0 & 0 \\
        0 & 1 - \lambda & 1 \\
        0 & 1 & 1 - \lambda
    \end{bmatrix} 
\end{equation*}

\begin{equation*}
    \det(A-\lambda I_3) = (3-\lambda) \cdot \begin{bmatrix} 1-\lambda & 1 \\ 1 & 1-\lambda \end{bmatrix} - 0 \cdot \begin{bmatrix} 0 & 0 \\ 0 & 1- \lambda \end{bmatrix} + 0 \cdot \begin{bmatrix} 0 & 1- \lambda \\ 0 & 1 \end{bmatrix} = (3-\lambda) \cdot ((1-\lambda)\cdot (1-\lambda) - (1\cdot 1))
\end{equation*}
Definition 21.12~\cite{NUMLA}
\begin{equation*}
    (3-\lambda) \cdot ((1-\lambda)\cdot (1-\lambda) - (1\cdot 1)) = (3-\lambda) \cdot (\lambda^2 - 2\lambda)
\end{equation*}
Conclusion:
\begin{equation*}
    \det(A - \lambda I_3) = (3-\lambda) \cdot (\lambda^2 - 2\lambda)
\end{equation*}
\subsection*{(b-c) Some Eigenvalues}
Show that $0$ and $3$ are eigenvalues of $A$ and other values are welcome too.\\\\
The eigenvalues $\lambda$ can be found by finding the solutions to the characteristic polynomial from (a), since any $\lambda$ that results in the polynomials being 0 is an eigenvalue for $A$.\\
This means solving:
\begin{equation*}
    (3-\lambda) \cdot (\lambda^2 - 2\lambda) = 0
\end{equation*}
(With a hint from MatLab)  this can formulated to finding a solution to:
\begin{equation*}
    3-\lambda = 0 \quad \text{or} \quad \lambda^2 - 2\lambda = 0
\end{equation*}
trivially a solution is $\lambda = 3$, making 3 an eigenvalue for $A$:
\begin{equation*}
    3 - \lambda = 0 = 3 - 3
\end{equation*}
For the other one $\lambda^2 - 2\lambda$, the quadratic formula can be used\\
 to find the roots for $1 \cdot \lambda^2 - 2 \cdot \lambda + 0 = 0 $.
 \begin{equation*}
    \lambda = \frac{-(-2) \pm \sqrt{(-2)^2 - 4 \cdot 1 \cdot 0}}{2 \cdot 1} = \frac{2 \pm \sqrt{4}}{2} = \frac{2 \pm 2}{2}
 \end{equation*}
This leaves two solutions
\begin{equation*}
    \frac{2 + 2}{2} = 2 \quad  \text{and} \quad \frac{2 - 2}{2} = 0
\end{equation*}
which means the the eigenvalues for $A$ is $ [3,2,0]$
\subsection*{(d) Basis using Eigenvectors}
Determine a basis of $\mathbb{R}^3$ consisting of eigenvectors of $A$.\\\\
Proposition 21.5~\cite{NUMLA}, says that if the eigenvalues of $A$ are unique and there is a set of eigenvectores for $A$, then the eigenvectores constitues a basis for $A$.\\
Eigenvectores can be found by solving for every $v$ in $(A - \lambda I_3)v = 0$ using the eigenvalues $\lambda \in [3,2,0]$\\\\
