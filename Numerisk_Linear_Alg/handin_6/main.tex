\section*{Math done by a human}
Consider the space of differentiable functions $f:[0,\pi] \to \mathbb{R}$ with the $L^2$ inner product
\begin{equation} \label{l2}
    \langle f,g \rangle = \int_{0}^{\pi} f(x)g(x) \, dx.
\end{equation}







\subsection*{Showing Orthogonality}
For the given collection of functions defined on $x \in [0,\pi]$
\begin{equation} \label{col}
    \sin(x),\sin(2x),\hdots ,\sin(kx), \quad \text{For every integer } k > 0
\end{equation}

\noindent Using equation\eqref{l2} and applying the collection\eqref{col}, it's possible to express the inner product of collection as such:
\begin{equation}
    \int_{0}^{\pi} \sin(nx)\sin(mx) \, dx \quad \text{for }m \neq n 
\end{equation}
note the $m\neq n$ is used since $ \langle f,f \rangle$ is the norm.\\
This integral can be made simpler by using the trigonometric identity 9.15~\cite{NUMLA}.
\begin{equation*}
     \int_{0}^{\pi} \frac{1}{2}[\cos((n-m)x)-\cos((n+m)x)] \, dx \quad \text{for }m \neq n 
\end{equation*}
then using the integral rules, by taking the constants out and splitting the integral.
\begin{equation} \label{changed}
    \frac{1}{2}[
        \int_{0}^{\pi} \cos((n-m)x) \, dx - \int_{0}^{\pi} \cos((n+m)x) \, dx
    ]
\end{equation}
 from here each integral can be evaluated, starting with the first one.\\
 Considering the first integral from equation\eqref{changed}, $\cos((n-m)x)$, this can be seen as a composite function on the form $h(i(x))$, where $h(u) = \cos(u)$ and $i(x) = (n-m)x$.\\
 This means the integral can be evaluated by substitution.\\ 
 Let $u = (n-m)x$ and $du = (n-m)dx$, then express $dx$ as a product of $u$.
 \begin{align*}
    &du = (n-m)dx \\
    &dx = \frac{1}{(n-m)}du
 \end{align*}
Note the limits $\int_{0}^{\pi}$ also change to accomodate $u$, this is done by putting the value of the limits into $i(x)$, which gives.
\begin{align*}
    &i(\pi) = (n-m)(\pi) = \pi(n-m) \\
    &i(0) = (n-m)(0) = 0
\end{align*}
then substitute the new limits into the respective integral that was chosen.
\begin{equation*}
    \int_{0}^{\pi} \cos((n-m)x) \, dx \to \int_{0}^{\pi(n-m)} \cos(u) \frac{1}{(n-m)} \, du
\end{equation*}
remember to take the constant out:
\begin{equation} \label{single}
    \frac{1}{(n-m)} \int_{0}^{\pi(n-m)} \cos(u) \, du
\end{equation}
now to integrate and evaluate \eqref{single}.
\begin{align*}
    &\frac{1}{(n-m)} [\sin(u)]_{0}^{\pi(n-m)}  \\
 =\, &\frac{1}{(n-m)}(\sin((n-m)\pi) - \sin(0)) \\
 = \, & \frac{1}{(n-m)}(0-0) \\
 = \, & 0
\end{align*}
The reason for this is that $n,m\in k$ (integers) and $\sin(\pi) = 0$. Since any whole multiplier of $\pi$ is either a half (odd number) or whole (even number) rotation which, in Mr.$\sin()$ eyes, is still $0$.\\
It's assumed $\sin(0)$ needs no explanation.\\\\
With all that in order, the same can be shown for the second integral in equation\eqref{changed} resulting in:
\begin{equation*}
    \frac{1}{2}[0-0] = 0
\end{equation*}
which means the inner product of the collection\eqref{col} is $0$ meaning it's an orthogonal collection.
\subsection*{Constant function $f(x)$}
Given the constant funnction $f(x) \equiv 1$, how does that change the orthogonality when $k$ is 'even' or 'odd'?.\\
Inserting the new $f(x)$ in equation\eqref{l2}.
\begin{equation*}
    \langle f,g \rangle = \int_{0}^{\pi} 1 \cdot g(x) \, dx.
\end{equation*}
this prevents the use of trigonominal identities resultig in evaluating $\cos()$ instead of $\sin()$ in the integral.\\
By evaluating the integral via substitution, like in the previous integral.
\begin{align*}
    & \int_{0}^{k\pi} \sin(u) \frac{1}{k}\, du\\
    & = \frac{1}{k} \int_{0}^{k\pi} \sin(u) \, du\\
    & = \frac{1}{k}[-\cos(u)]_{0}^{k\pi} \\
    & = \frac{1}{k}(-\cos(k\pi)-(-\cos(0))) \\
    & = \frac{1}{k} (\cos(0)-\cos(k\pi))\\
    & = \frac{1}{k}(1 - \cos(k\pi))
\end{align*}
And now for the magic!
$\frac{1}{k}(1 - \cos(k\pi)) =
\begin{cases}
    0 \quad \text{for an even } k   \\
    \frac{2}{k} \quad \text{for an odd } k
\end{cases}
$
which means if $k$ is odd it isn't orthogonal and if it's even it is orthogonal.\\
So for $\sin(x),\,\sin(3x)$ isn't orthogonal and $\sin(2x),\, \sin(4x)$ is orthogonal.





\subsection*{Projection}
Determinign the projection onto $\sin(x)$ and $\sin(3x)$ using $f(x)\equiv 1$ yiels:
\begin{equation} \label{proj1}
    proj_{\sin(x)}(1) = \frac{\int_{0}^{\pi} 1 \cdot \sin(x) \, dx}{\int_{0}^{\pi}\sin(x)\cdot\sin(x)\,dx} \sin(x)   
\end{equation}
\begin{equation} \label{proj3}
    proj_{\sin(3x)}(1) = \frac{\int_{0}^{\pi} 1 \cdot \sin(3x) \, dx}{\int_{0}^{\pi}\sin(3x)\cdot\sin(3x)\,dx} \sin(3x)
\end{equation}
Using the results from the previous section, the numerator in\eqref{proj1} and\eqref{proj3} can be generalized to $\frac{2}{k}$ since both $\sin(x)$ and $\sin(3x)$ use odd $k$ values.\\
Before inserting anything the denominator shall be evaluated, starting with equation\eqref{proj1}.
\begin{equation*}
    \int_{0}^{\pi}\sin(x)\cdot\sin(x)\,dx
\end{equation*}
The above integral can be simplified using the trigonometric identities 9.15~\cite{NUMLA} but for $\alpha = \beta$ and $\alpha = x$
\[
2\sin^2(x) = -\cos(2x) + \cos(0) 
\]
isolate $\sin^2(x)$ to fit the case, remember $\cos(0) = 1$
\begin{equation} \label{deno}
    \sin^2(x) = \frac{1-\cos(2x)}{2}
\end{equation}
substituting that into the integral
\begin{equation*}
    \int_{0}^{\pi} \frac{1-\cos(2x)}{2}
\end{equation*}
extracting the constant and evaluating gets the following
\begin{equation*}
    \frac{1}{2}[x-\sin(2x)]_{0}^{\pi}
\end{equation*}
when evaluated
\begin{equation} \label{dres}
    \frac{1}{2}((\pi-0)-(0-0)) = \frac{1}{2}\pi = \frac{\pi}{2}
\end{equation}
It's important to note for this, that the result from integrating equation\eqref{deno} using either $\sin^2(x)$, as was just done, or $\sin^2(3x)$ would yield the same result $\frac{\pi}{2}$ since $\sin(2x)$ and $\sin(6x)$ both equal $0$.\\
By combining the findings for the denominator\eqref{dres} and the numerator(see start of section \textbf{Projection}), 
the equations\eqref{proj1}\eqref{proj3} can be rewritten as:
\begin{equation} \label{c1}
    proj_{\sin(x)}(1) = \frac{2}{\frac{\pi}{2}} \sin(x) = \frac{4}{\pi} \sin(x)
\end{equation}
 \begin{equation} \label{c3}
    proj_{\sin(3x)}(1) = \frac{\frac{2}{3}}{\frac{\pi}{2}} \sin(3x) = \frac{4}{3\pi} \sin(3x)
\end{equation}
these projections represents (mathematically) the place on the sine waves the looks like the line $f(x) \equiv 1$ the most. According to equation (13.3)~\cite{NUMLA} adding the projections together will
result in a better approximation of line on the span.\\\\

\noindent The given $g(x) = 1 + c_1 \sin(x) + c_3 \sin(3x)$ looks a bit like the (c) from proposition 13.10~\cite{NUMLA}, the residual vector.
\begin{equation}
    f(x) - ( proj_{\sin(x)}(1)+proj_{\sin(3x)}(1)) \perp w
\end{equation}
and remember $f(x) \equiv 1$, so the difference is that $g(x)$ adds the projected vector $P$ to the original vector $V$ (see cited proposition for notation) instead of subtracting it, which wouldn't make the resulting vector orthogonal to the defined span [$\sin(x),\sin(3x)$] (trust).\\
So to make $g(x)$ orthogonal to the span, the sign has to be cancelled out, this can be done by 'reversing' the constants $c_1$ and $c_3$ from equation\eqref{c1}\eqref{c3}, note the $\sin()$ isn't part of the constants.
\begin{equation}
    \frac{3}{\pi} + \frac{4}{3\pi} \to (-\frac{3}{\pi}) - \frac{3}{3\pi}
\end{equation}
this should cancel out the '+' sign in $g(x)$ and make it orthogonal, which can be shown by evaluating the inner product $\langle g(x),\sin(x) \rangle$.
\begin{equation*}
    \langle g(x),\sin(x) \rangle = \int_{0}^{\pi} (1 + c_1\sin(x) + c_3\sin(3x)) \cdot \sin(x) \, dx
\end{equation*}
Looking inside the integral, it can be simplifed to:
\[

\[
\int_{0}^{\pi} \sin(x) + c_1\sin^2(x) \, dx

\[
\int_{0}^{\pi} \sin(x) \, dx + c_1 \int_{0}^{\pi} \sin^2(x) \, dx
